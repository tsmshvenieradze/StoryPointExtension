    ```typescript
    // src/audit/parse.ts — Source: D-04, D-06, D-08, D-09, D-10, D-11, D-12, D-17;
    // regex verified against all D-23 cases live in Node 24.15
    import type { Level } from '../calc/levels';
    import { LEVELS } from '../calc/levels';
    import type { FibonacciSp } from '../calc/fibonacci';
    import type { AuditPayload } from './types';

    const FIB_VALUES: ReadonlySet<FibonacciSp> = new Set<FibonacciSp>([0.5, 1, 2, 3, 5, 8, 13]);
    const SENTINEL_RX = /<!--\s*sp-calc:v1\s+(\{[^{}]*\})\s*-->/;

    /** Replace ONLY non-breaking space (U+00A0) with ASCII space.
     *  Do NOT collapse other whitespace — internal JSON whitespace is preserved
     *  and JSON.parse handles it natively. (RESEARCH Pitfall 2.) */
    function normalizeNbsp(s: string): string {
      return s.replace(/ /g, ' ');
    }

    /** Title-Case lookup — case-insensitive match against LEVELS. (D-04) */
    function toCanonicalLevel(input: unknown): Level | null {
      if (typeof input !== 'string') return null;
      const lc = input.toLowerCase();
      for (const lvl of LEVELS) {
        if (lvl.toLowerCase() === lc) return lvl;
      }
      return null;
    }

    export function parse(commentBody: string): AuditPayload | null {
      if (typeof commentBody !== 'string' || commentBody.length === 0) return null;
      const normalized = normalizeNbsp(commentBody);
      const match = normalized.match(SENTINEL_RX);
      if (!match || !match[1]) return null;

      let raw: unknown;
      try {
        raw = JSON.parse(match[1]);
      } catch {
        return null;  // D-12: never throw on user input
      }

      if (typeof raw !== 'object' || raw === null) return null;
      const obj = raw as Record<string, unknown>;

      // schemaVersion strictly === 1 per D-06, D-08
      if (obj['schemaVersion'] !== 1) return null;

      // sp must be a known Fibonacci value
      const sp = obj['sp'];
      if (typeof sp !== 'number' || !FIB_VALUES.has(sp as FibonacciSp)) return null;

      // c/u/e: case-insensitive lookup per D-04
      const c = toCanonicalLevel(obj['c']);
      const u = toCanonicalLevel(obj['u']);
      const e = toCanonicalLevel(obj['e']);
      if (c === null || u === null || e === null) return null;

      return { sp: sp as FibonacciSp, c, u, e, schemaVersion: 1 };
    }
    ```

    - Do NOT use `.+?` (non-greedy any-char) instead of `[^{}]*` — Pitfall 3

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| `parse(commentBody: string)` input | The single runtime trust boundary in Phase 1. `commentBody` is opaque user-controlled text that originated as the `text` field of an ADO `Comment` object — this can include user-typed content from the work-item comment editor, ADO renderer markup wrapping the original sentinel comment, copy-paste from rich-text sources (NBSP, ZWSP risk), and adversarial edits to the JSON inside the sentinel marker. The parser MUST treat every input as untrusted. |
| Other audit functions (`serialize`, `parseLatest`) | Receive only TypeScript-typed inputs (`AuditPayload`, `ReadonlyArray<AdoComment>`); compile-time correctness ensures the inputs are well-formed unless the caller bypasses TypeScript. `parseLatest` delegates to `parse` for actual untrusted-text handling. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01-A1 | Tampering | `parse()` accepts an attacker-edited sentinel where `schemaVersion` is bumped to 2 to bypass field validation | mitigate | `if (obj['schemaVersion'] !== 1) return null` (parse.ts) — strict equality rejects any value other than literal `1`; verified by Task 7 edge case rows 'schemaVersion=2 returns null' and 'schemaVersion=0 returns null'. The forward-incompatibility is intentional (D-06, D-07): a v2 future schema with a different shape is silently dropped rather than misinterpreted. |
| T-01-A2 | Tampering | `parse()` accepts an attacker-edited `sp` value outside the canonical Fibonacci set (e.g., 7, 100, 0) leading downstream consumers to display arbitrary numbers | mitigate | `FIB_VALUES.has(sp as FibonacciSp)` (parse.ts) — Set membership check; only {0.5, 1, 2, 3, 5, 8, 13} pass. Verified by Task 7 edge case 'unknown sp value (4 — not Fibonacci) returns null'. |
| T-01-A3 | Tampering | `parse()` accepts an attacker-edited level label outside the canonical 5 (e.g., "Trivial", "Critical") | mitigate | `toCanonicalLevel` does case-insensitive lookup against `LEVELS`; returns null if no match. Verified by Task 7 edge case 'unknown label (Trivial) returns null'. The canonicalization to Title Case is honest about the lookup result — the AuditPayload always contains a member of the closed Level union. |
| T-01-A4 | Denial of Service | Catastrophic regex backtracking when parser receives a 1MB comment body crafted to exploit `[^{}]*` or any other regex construct | mitigate | The regex `/<!--\s*sp-calc:v1\s+(\{[^{}]*\})\s*-->/` uses a NEGATED character class `[^{}]*` which is bounded — it cannot backtrack across `{` or `}` characters. Each character is examined at most once per run. Worst-case behavior on a 1MB input is O(n) linear scan. RESEARCH §Pattern 5 explicitly chose this over the non-greedy `.+?` alternative for backtracking robustness. (`.+?` with the `s` flag CAN exhibit polynomial backtracking under pathological input; `[^{}]*` cannot.) |
| T-01-A5 | Denial of Service | `JSON.parse` on a deeply-nested or oversized JSON string causes stack overflow / memory exhaustion | accept | The regex's `[^{}]*` already excludes any input with interior `{` or `}` characters. Therefore the JSON the parser ever sees is structurally flat (no nested objects or arrays) and bounded by the comment body's length. ADO's comment size limit is documented at ~32KB; even at the cap, a flat JSON parse is well within Node's safe operating envelope. The threat is materially impossible given the regex's filtering. |
| T-01-A6 | Information Disclosure | `parse` leaks information about the regex internals or stack frames via thrown exceptions | mitigate | The `try { JSON.parse(...) } catch { return null }` block catches any throw from `JSON.parse`. The function is otherwise side-effect-free (no console output, no network, no filesystem). All error paths return `null` per D-12; no exception propagates to the caller. Verified by Task 7's 'malformed JSON inside sentinel returns null' case. |
| T-01-A7 | Tampering | Prototype pollution via `JSON.parse` of `{"__proto__": {"polluted": true}}` inside the sentinel | mitigate | Two layers of defense: (1) the `[^{}]*` regex rejects nested-object JSON, and a `__proto__` payload would require a nested object to be exploitable; a flat `{"__proto__": "string"}` does not pollute Object.prototype because it's just a string-valued own property. (2) After `JSON.parse`, the parser performs explicit shape validation key-by-key (`obj['schemaVersion']`, `obj['sp']`, `obj['c']`, etc.) and constructs a NEW object literal `{ sp, c, u, e, schemaVersion: 1 }` rather than returning the raw parse result. The returned `AuditPayload` does not retain any prototype mutation that might have occurred during parsing. |
| T-01-A8 | Tampering | `parseLatest` mutates the caller's input array, leading to surprising behavior in Phase 4 ADO bridge if it later iterates the array for another purpose | mitigate | `[...comments].sort(...)` (parseLatest.ts) creates a shallow copy before sorting. Verified by Task 8's 'does not mutate the input array' test which captures `[...arr]` before calling `parseLatest` and asserts `arr` deep-equals the snapshot afterward. |
| T-01-A9 | Elevation of Privilege | Audit module imports browser-only or SDK APIs by mistake, breaking the zero-deps invariant | mitigate | Task 9 hard-greps for `react`, `react-dom`, `azure-devops-extension-sdk`, `azure-devops-extension-api`, `window`, `document`, `navigator`, `crypto.subtle` and asserts zero hits across `src/audit/**`. D-25 and D-26 are mechanically enforced. |
| T-01-A10 | Tampering | `JSON.stringify` with a future code refactor silently changes wire format because of insertion-order dependency | mitigate | Replacer-array form `JSON.stringify(payload, ['sp','c','u','e','schemaVersion'])` (serialize.ts) enforces key order independent of object literal authoring order. Verified by Task 6's 'produces identical output regardless of input field declaration order' test. RESEARCH §Pitfall 1 documents this in detail. |

**Notes on accepted risks:**
- **A8 in Plan 01-01 (NBSP only)**: ZWSP (U+200B) and other invisible Unicode are NOT normalized. If a user pastes ZWSP between marker and JSON, the regex's `\s+` already tolerates it (ZWSP matches `\s` in V8). If ZWSP appears inside the JSON, `JSON.parse` rejects it and we return null. Both behaviors are correct.
- **HTML escaping**: Assumption A8 from RESEARCH (the sentinel `<!-- ... -->` survives ADO's renderer without `&lt;!--` escaping) is a Phase 2 verification gate. If Phase 2 surfaces escaping in some renderer mode, a 5-line parser update tolerates both prefixes — does not invalidate the threat model.
